void Arena::drawArena() {
	std::string upper_and_lower_borders(size[1], '-');
	int space = 1;
	if (size[1] > 9) {
		space = 2;
	}
	else if (size[1] > 99) {
		space = 3;
	}
	std::cout << std::string(space + 2, ' ') << upper_and_lower_borders << std::endl;
	for (int col = 0; col < size[0]; col++) {
		if (col < 9) {
			std::cout << col + 1 << std::string(space, ' ') << '|';
		}
		else if (col < 99) {
			std::cout << col + 1 << std::string(space - 1, ' ') << '|';
		}
		else {
			std::cout << col + 1 << std::string(space - 2, ' ') << '|';
		}
		for (int row = 0; row < size[1]; row++) {
			std::cout << playingField[col][row];
		}
		std::cout << "|" << std::endl;
	}
	std::cout << std::string(space + 2, ' ') << upper_and_lower_borders << std::endl;

}
